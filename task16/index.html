const expenseList = document.getElementById('expense-list'); const totalBudgetDisplay = document.getElementById('total-budget'); let totalBudget = 0; const expenses = {}; // Store expenses by category expenseForm.addEventListener('submit', (event) => { event.preventDefault(); const category = document.getElementById('category').value; const amount = parseFloat(document.getElementById('amount').value); const description = document.getElementById('description').value; // Input validation (ensure all fields are filled) if (!category || !amount || !description) { alert('Please fill in all fields.'); return; } // Update expenses and total budget if (!expenses[category]) { expenses[category] = []; } expenses[category].push({ amount, description }); totalBudget += amount; // Update UI updateExpenseList(); totalBudgetDisplay.textContent = totalBudget.toFixed(2); expenseForm.reset(); // Clear the form }); function updateExpenseList() { expenseList.innerHTML = ''; // Clear previous list for (const category in expenses) { const categoryList = document.createElement('ul'); const categoryHeader = document.createElement('li'); categoryHeader.textContent = `${category} ($${calculateCategoryTotal(category).toFixed(2)})`; categoryList.appendChild(categoryHeader); expenses[category].forEach(expense => { const expenseItem = document.createElement('li'); expenseItem.textContent = `- $${expense.amount.toFixed(2)}: ${expense.description}`; categoryList.appendChild(expenseItem); }); expenseList.appendChild(categoryList); } } function calculateCategoryTotal(category) { return expenses[category].reduce((total, expense) => total + expense.amount, 0); } ``` **Key Improvements:** - **Clearer Structure:** HTML is organized into sections for the form, budget summary, and expense list. - **Input Validation:** JavaScript ensures all form fields are filled before adding an expense. - **Category Grouping:** Expenses are grouped by category, with subtotals for each category. - **Dynamic Updates:** The expense list and total budget are updated automatically when new expenses are added.
